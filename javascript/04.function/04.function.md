## 문법
**[함수]**
1. 선언
2. 표현식

```javascript
function 함수이름(매개변수) {
  // 함수 본문. return 값이 없으면 undefined 반환

  // 에러 직접 발생시킬 경우
  if (조건1 || 조건2) {  // Truthy할 경우 실행
    throw new Error(
      '에러 문구 입력'
    );
  }

  return ...;
}

 // 함수를 변수에 할당 후 출력 또는 바로 출력 통해 결과값 확인
let 변수 = 함수이름(인자);  // 함수에 전달
console.log(변수);
```
